function



    const fruits = ['apple', 'banana', 'cherry'];
    const vegetables = ['tomato', 'cucumber', 'potato', 'onion'];
    const animals = ['cat', 'dog'];
    //DRY code?
    fruits.forEach(x => console.log(x));
    vegetables.forEach(x => console.log(x));
    animals.forEach(x => console.log(x));
    // DRY code
    function logItems (items) {
        items.forEach(x => console.log(x));
    }

    logItems(fruits);
    logItems(vegetables);
    logItems(animals);
